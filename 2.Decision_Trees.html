def evaluacion(model, X_test, y_test):
    metricas = {}
    # Predecimos los valores de 'Precio' para el conjunto de prueba
    y_pred = model.predict(X_test)
    
    # MSE - Error Cuadrtico Medio
    metricas['MSE'] = mean_squared_error(y_test, y_pred)

    # RMSE - Raz del Error Cuadrtico Medio
    metricas['RMSE'] = np.sqrt(metricas['MSE'])

    # MAE - Error Absoluto Medio
    metricas['MAE'] = mean_absolute_error(y_test, y_pred)

    # MedAE - Mediana del Error Absoluto
    metricas['MedAE'] = median_absolute_error(y_test, y_pred)

    # Coeficiente de Determinacin, R
    metricas['R2'] = r2_score(y_test, y_pred)

    # Imprimir resultados
    print("MSE:", metricas['MSE'])
    print("RMSE:", metricas['RMSE'])
    print("MAE:", metricas['MAE'])
    print("MedAE:", metricas['MedAE'])
    print("R:", metricas['R2'])
    
    return metricas

def comparar_metricas(modelos_metricas, nombres_modelos):
    metricas = ['MSE', 'RMSE', 'MAE', 'MedAE', 'R2', 'MAPE', 'Tiempo de entrenamiento']
    
    data = {metrica: [modelo[metrica] for modelo in modelos_metricas] for metrica in metricas}
    df = pd.DataFrame(data, index=nombres_modelos)

    # Iteramos sobre cada mtrica y crear una grfica de barras
    for metrica in metricas:
        # Obtenemos los valores para la mtrica actual de cada modelo
        valores = [modelo[metrica] for modelo in modelos_metricas]

        # Creamos la grfica de barras
        plt.figure(figsize=(10, 6))
        plt.bar(nombres_modelos, valores, color=['blue', 'green', 'red', 'purple'])
        plt.title(f'Comparacin de {metrica}')
        plt.ylabel(metrica)
        plt.xlabel('Modelos')
        for i, v in enumerate(valores):
            plt.text(i, v + max(valores)*0.01, f"{v:.2f}", ha='center')
        plt.show()
        
    return df

def sobreajuste(model, X_train=X_train, X_test=X_test, y_train=y_train, y_test=y_test):
    y_pred = model.predict(X_test)
    
    # Obtenemos el error en cada iteracin
    test_score = [mean_squared_error(y_test, y_pred) for y_pred in model.staged_predict(X_test)]
    train_score = [mean_squared_error(y_train, y_pred) for y_pred in model.staged_predict(X_train)]
    
    # Creamos grfico
    plt.figure(figsize=(12, 6))
    plt.plot(train_score, label='Training Error')
    plt.plot(test_score, label='Validation Error')
    plt.title('Error de Entrenamiento vs. Error de Validacin')
    plt.xlabel('Nmero de rboles')
    plt.ylabel('Error Cuadrtico Medio')
    plt.legend()
    plt.show()

def objective_rf(space, X_train, y_train, X_val, y_val):
    model = RandomForestRegressor(**space)
    # Realizamos validacin cruzada y calcular el error medio
    kf = KFold(n_splits=5, shuffle=True, random_state=42)
    mae_scores = cross_val_score(model, X, y, cv=kf, scoring='neg_mean_absolute_error')
    mean_mae = -np.mean(mae_scores)  # Convertimos a positivo
    return {'loss': mean_mae, 'status': STATUS_OK}

def objective_gb(space, X_train, y_train, X_val, y_val):
    model = GradientBoostingRegressor(**space)
    # Realizamos validacin cruzada y calcular el error medio
